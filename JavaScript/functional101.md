# Functional Programming in JS

- Don't alter any variables
- Return new variables
- Side effects are 
- Declare function arguments _any computation inside a function depends only on the arguments, and not on any global object or variable_


```JS
// The global variable
// this global variable does not get changed with these functions below
var bookList = ["The Hound of the Baskervilles", "On The Electrodynamics of Moving Bodies", "PhilosophiÃ¦ Naturalis Principia Mathematica", "Disquisitiones Arithmeticae"];

function add (arr, bookName) {
  // this copies the array passed to it, and appends the final item
  return [...arr, bookName];
}

function remove (arr, bookName) {
  // this returns the passed in array, but removes any items that match what was passed
  return arr.filter(book => book !== bookName);
}

var newBookList = add(bookList, 'A Brief History of Time');
var newerBookList = remove(bookList,'On The Electrodynamics of Moving Bodies');
var newestBookList = remove(add(bookList, 'A Brief History of Time'), 'On The Electrodynamics of Moving Bodies');

console.log(bookList);

```

- Building a filter method from scratch
  - Shown two ways to do it, with forEach and a for loop 
```JS
// The global variable
var s = [23, 65, 98, 5];

Array.prototype.myFilter_forEach = function(callback){
  var newArray = [];

  // old school 
  // this.forEach(function (element) {if(callback(element) === true) newArray.push(element)})
  // ES6
  this.forEach(element => callback(element) ? newArray.push(element):false);
  return newArray;

};

Array.prototype.myFilter_forLoop = function(callback) {
  var newArray = [];
  for (let i = 0; i < this.length; i++) {
    if (callback(this[i]) === true) {
      newArray.push(this[i]);
    }
  }
  return newArray;
};


// using the first method to filter
var new_s = s.myFilter_forEach(function(item){
  return item % 2 === 1;
});


console.log(new_s)
console.log(s)
```

# Slice

Returns a copy of an array with an optional starting and ending index (non-inclusive)
`array.slice(start, end);`


```JS
function sliceArray(anim, beginSlice, endSlice) {
  let newArr = [];
  newArr = anim.slice(beginSlice, endSlice);
  return newArr;

}
var inputAnim = ["Cat", "Dog", "Tiger", "Zebra", "Ant"];
console.log(sliceArray(inputAnim, 1, 3));
```
